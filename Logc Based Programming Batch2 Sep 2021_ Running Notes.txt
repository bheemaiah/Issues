LBP1

Program to check whether the given number is even or odd number.

inputs ----------> a number from the user
output  ---------> even or odd or invalid
constraint ------> n>=0

Logic:
~~~~~~
1. read a number from the user. (scanf,nextInt,int(input()))
2. if n>=0 then
	2.1: if n%2==0 then print even number
	2.2: else print odd number
3. else print invalid

C Implementation:
~~~~~~~~~~~~~~~~~
#include <stdio.h>
#include <string.h>
#include <math.h>
#include <stdlib.h>

int main() {
    int n; //declaration of an int
    scanf("%d",&n);//read n value
    if(n>=0){
        if(n%2==0)
            printf("even");        
        else
            printf("odd");        
    }
    else
        printf("invalid");    
    return 0;
}

        Scanner obj = new Scanner(System.in);

            //logic
            if(n%2==0)
                System.out.println("even");
            else
                System.out.println("odd");
        }
        else
            System.out.println("invalid");
    }
}

constraint --------> n>0

123 -----> 3 2 1

C Implementation:
~~~~~~~~~~~~~~~~~
#include <stdio.h>
#include <string.h>
#include <math.h>
#include <stdlib.h>

int main() {
    int n,d;
    scanf("%d",&n);
    while(n!=0){
        d=n%10;
        printf("%d ",d);
        n=n/10;
    }
    return 0;
}

Java Implementation:

import java.io.*;
import java.util.*;

public class Solution {

    public static void main(String[] args) {
        Scanner obj = new Scanner(System.in);
        int n = obj.nextInt(),d;
        while(n!=0){
            d=n%10;
            System.out.print(d+" ");
            n=n/10;
        }
    }
}

int max(int a,int b,int c){
